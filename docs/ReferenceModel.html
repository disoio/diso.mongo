  dereference : (callback)->
    @model.constructor.find(
      query : {
        _id : @_id
      }
      callback : (error, model)=>
        if error 
          callback(error)
        else 
          if model
            @dereferenced = true
            @model = model
            callback(null, model)
          else
            callback("Dereference failed")
    )

  data : (args...)->
    @model.data.apply(@model, args)

  deflate : ()->
    @model._map(
      $model : true
